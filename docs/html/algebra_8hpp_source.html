179  inline Eigen::Matrix3d MatrixLog3(const Eigen::Matrix3d& R) {
180  double acosinput = (R.trace() - 1) / 2.0;
181  double theta = std::acos(acosinput);
182  Eigen::MatrixXd m_ret = Eigen::MatrixXd::Zero(3, 3);
183  if (acosinput >= 1)
184  return m_ret;
185  else if (acosinput <= -1) {
186  Eigen::Vector3d omg;
187  if (!Epsilon(1 + R(2, 2)))
188  omg = (1.0 / std::sqrt(2 * (1 + R(2, 2))))*Eigen::Vector3d(R(0, 2), R(1, 2), 1 + R(2, 2));
189  else if (!Epsilon(1 + R(1, 1)))
190  omg = (1.0 / std::sqrt(2 * (1 + R(1, 1))))*Eigen::Vector3d(R(0, 1), 1 + R(1, 1), R(2, 1));
191  else
192  omg = (1.0 / std::sqrt(2 * (1 + R(0, 0))))*Eigen::Vector3d(1 + R(0, 0), R(1, 0), R(2, 0));
193  m_ret = VecToso3(M_PI * omg);
194  //m_ret = VecToso3(theta * omg);
195  return m_ret;
196  }
197  else {
198  m_ret = theta * (R - R.transpose())/(2.0*sin(theta));
199  return m_ret;
200  }

224  inline Eigen::MatrixXd
225  MatrixExp6(const Eigen::MatrixXd& se3mat) {
226  // Extract the angular velocity vector from the transformation matrix
227  Eigen::Matrix3d se3mat_cut = se3mat.block<3, 3>(0, 0);
228  Eigen::Vector3d omgtheta = so3ToVec(se3mat_cut);
229 
230  Eigen::MatrixXd m_ret(4, 4);
231 
232  // If negligible rotation, m_Ret = [[Identity, angular velocty ]]
233  // [[0 , 1]]
234  if (Epsilon(omgtheta.norm())) {
235  // Reuse previous variables that have our required size
236  se3mat_cut = Eigen::MatrixXd::Identity(3, 3);
237  omgtheta << se3mat(0, 3), se3mat(1, 3), se3mat(2, 3);
238  m_ret << se3mat_cut, omgtheta,
239  0, 0, 0, 1;
240  return m_ret;
241  }
242  // If not negligible, MR page 105
243  else {
244  double theta = (AxisAng3(omgtheta))(3);
245  Eigen::Matrix3d omgmat = se3mat.block<3, 3>(0, 0) / theta;
246  Eigen::Matrix3d expExpand = Eigen::MatrixXd::Identity(3, 3) * theta + (1 - std::cos(theta)) * omgmat + ((theta - std::sin(theta)) * (omgmat * omgmat));
247  Eigen::Vector3d linear(se3mat(0, 3), se3mat(1, 3), se3mat(2, 3));
248  Eigen::Vector3d GThetaV = (expExpand*linear) / theta;
249  m_ret << MatrixExp3(se3mat_cut), GThetaV,
250  0, 0, 0, 1;
251  return m_ret;
252  }
